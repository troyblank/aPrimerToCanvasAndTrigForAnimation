function drawBall(){
    clearCanvas(canvasContext);
    canvasContext.beginPath();
    canvasContext.arc(xVal, yVal, 20, 0, Math.PI*2, false);
    canvasContext.closePath();
    canvasContext.fill();
}

function startBall(){
    timer = setInterval(animateBall, 33);
}

function animateBall(e){
    xVal += xVelocity;
    yVal += yVelocity;
    drawBall();
}

function clearCanvas(cxt) {
    //clears all pixels on the canvas in the given rectangle to transparent.
    cxt.clearRect(0, 0, cxt.canvas.width, cxt.canvas.height);
}
