Let's add the math to our animation

    
var canvasContext;
var angle = 0;
var xVal = 0;
var yVal = 0;
var xCenter = 300;
var yCenter = 150;
var angleVelocity = 0.1;
var radius = 100;
var timer;
        
function animationExampleInit(e){
    canvasContext = $("#myCanvas")[0].getContext("2d");
    drawBall();
    startBall();
}

function drawBall(){
    clearCanvas(canvasContext);
    canvasContext.beginPath();
    canvasContext.arc(xVal, yVal, 20, 0, Math.PI*2, false);
    canvasContext.closePath();
    canvasContext.fill();
}

function startBall(){
    timer = setInterval(animateBall, 33);
}

function animateBall(e){
    angle += angleVelocity;
    xVal = xCenter + Math.cos(angle) * radius;
    yVal = yCenter + Math.sin(angle) * radius;
    drawBall();
}
